/* ── Count-up ── */
(function(){
  var done=new Set();
  var obs=new IntersectionObserver(function(entries){
    entries.forEach(function(e){
      if(!e.isIntersecting||done.has(e.target))return;
      done.add(e.target);
      var el=e.target,to=parseInt(el.dataset.count),suf=el.dataset.suf||'';
      if(!to)return;
      var t0=Date.now(),dur=1600;
      (function run(){
        var p=Math.min((Date.now()-t0)/dur,1),ease=1-Math.pow(1-p,3);
        el.textContent=Math.floor(ease*to)+suf;
        if(p<1)requestAnimationFrame(run);
      })();
    });
  },{threshold:.5});
  document.querySelectorAll('[data-count]').forEach(function(el){obs.observe(el)});
})();

      ],{duration:dur,easing:'ease-out',fill:'forwards'}).finished.then(function(){
        p.remove();
      });
    }

    /* Auto-looping slider script inside IIFE */
    var slides = card.querySelectorAll('.prod-slider .slide');

          ['rgba(255,165,0,.6)','rgba(255,215,0,.5)','rgba(255,140,0,.4)'],
          ['rgba(230,162,164,.6)','rgba(180,200,120,.5)','rgba(248,215,217,.6)']

    if (slides.length) {
      var currentSlide = cardId === 'prod-bcaa' ? 2 : 0;

          if (i === index) {
            slide.classList.add('active');

            slide.classList.remove('active');

        });
        if (dots.length) {
          dots.forEach(function(dot, i) {
            if (i === index) {
              dot.classList.add('active');
            } else {
              dot.classList.remove('active');
            }
          });
        }
        currentSlide = index;
        if (swatchData) {
          cfg.colors = swatchData.colors[index];
        }
      }

      function nextSlide() {
        var next = (currentSlide + 1) % slides.length;
        showSlide(next);
      }

        slideInterval = setInterval(nextSlide, 3500);

        if (slideInterval) clearInterval(slideInterval);

      if (dots.length) {

            showSlide(idx);
          });
        });
      }

      // Initialize default active slide colors
      if (swatchData) {
        cfg.colors = swatchData.colors[currentSlide];
      }
    }

/* ── CONCEPT 4: TRI Fusion Molecule Sync Animation ── */
(function() {
  var canvas = document.getElementById('fusion-canvas');
  if (!canvas) return;
  var ctx = canvas.getContext('2d');
  var img = document.querySelector('.fusion-main-img');
  
  var w, h;
  function resize() {
    var rect = canvas.getBoundingClientRect();
    w = canvas.width = rect.width;
    h = canvas.height = rect.height;
  }
  resize();
  window.addEventListener('resize', resize);

  var numParticles = 48;
  var particles = [];
  var syncStartTime = 0;
  var state = 'floating'; // 'floating' | 'syncing' | 'orbiting'

  // Mouse interaction
  var mouse = { x: null, y: null, active: false, radius: 100 };
  var visContainer = document.querySelector('.fusion-vis');
  if (visContainer) {
    visContainer.addEventListener('mousemove', function(e) {
      var rect = visContainer.getBoundingClientRect();
      mouse.x = e.clientX - rect.left;
      mouse.y = e.clientY - rect.top;
      mouse.active = true;
    });
    visContainer.addEventListener('mouseleave', function() {
      mouse.x = null;
      mouse.y = null;
      mouse.active = false;
    });
  }

  // Particle constructor
  function Particle() {
    this.x = Math.random() * w;
    this.y = Math.random() * h;
    this.vx = (Math.random() - 0.5) * 1.5;
    this.vy = (Math.random() - 0.5) * 1.5;
    this.baseRadius = Math.random() * 2.5 + 1.5;
    this.radius = this.baseRadius;
    this.alpha = Math.random() * 0.4 + 0.3;
    // Rose gold & warm gold tones
    this.color = Math.random() > 0.4 ? 'rgba(230, 162, 164, ' : 'rgba(200, 169, 110, ';
    this.orbitRadius = Math.random() * 30 + 135;
    this.orbitAngle = Math.random() * Math.PI * 2;
    this.orbitSpeed = (Math.random() * 0.01 + 0.005) * (Math.random() > 0.5 ? 1 : -1);
  }

  for (var i = 0; i < numParticles; i++) {
    particles.push(new Particle());
  }

  // Scroll reveal trigger for sync
  var revealed = false;
  var isDrawing = false;
  var obs = new IntersectionObserver(function(entries) {
    entries.forEach(function(entry) {
      if (entry.isIntersecting) {
        if (!isDrawing) {
          isDrawing = true;
          draw();
        }
        if (!revealed && entry.intersectionRatio >= 0.2) {
          revealed = true;
          triggerSync();
        }
      } else {
        isDrawing = false;
      }
    });
  }, { threshold: [0, 0.2] });
  if (visContainer) obs.observe(visContainer);

  function triggerSync() {
    state = 'syncing';
    syncStartTime = Date.now();
    setTimeout(function() {
      if (img) {
        img.classList.add('visible');
      }
      state = 'orbiting';
    }, 1800);
  }

  function draw() {
    if (!isDrawing) return;
    ctx.clearRect(0, 0, w, h);
    var cx = w / 2;
    var cy = h / 2;
    var time = Date.now();

    for (var i = 0; i < particles.length; i++) {
      var p = particles[i];

      if (state === 'floating') {
        p.x += p.vx;
        p.y += p.vy;

        if (p.x < 0 || p.x > w) p.vx *= -1;
        if (p.y < 0 || p.y > h) p.vy *= -1;

      } else if (state === 'syncing') {
        var elapsed = (time - syncStartTime) / 1800;
        var targetX = cx + Math.cos(p.orbitAngle) * p.orbitRadius;
        var targetY = cy + Math.sin(p.orbitAngle) * p.orbitRadius;

        p.x += (targetX - p.x) * 0.08;
        p.y += (targetY - p.y) * 0.08;
        p.orbitAngle += p.orbitSpeed;

      } else if (state === 'orbiting') {
        p.orbitAngle += p.orbitSpeed;
        var targetX = cx + Math.cos(p.orbitAngle) * p.orbitRadius;
        var targetY = cy + Math.sin(p.orbitAngle) * p.orbitRadius;

        p.x += (targetX - p.x) * 0.12;
        p.y += (targetY - p.y) * 0.12;
      }

      // Mouse attraction
      if (mouse.active) {
        var dx = mouse.x - p.x;
        var dy = mouse.y - p.y;
        var dist = Math.sqrt(dx * dx + dy * dy);
        if (dist < mouse.radius) {
          var force = (mouse.radius - dist) / mouse.radius;
          p.x += dx * force * 0.05;
          p.y += dy * force * 0.05;
        }
      }

      var alpha = p.alpha;
      if (state === 'syncing') {
        var elapsed = Math.min((time - syncStartTime) / 1800, 1);
        alpha = p.alpha + (1 - p.alpha) * Math.sin(elapsed * Math.PI) * 0.4;
      }

      ctx.beginPath();
      ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
      ctx.fillStyle = p.color + alpha + ')';
      ctx.fill();

      ctx.beginPath();
      ctx.arc(p.x, p.y, p.radius * 2.5, 0, Math.PI * 2);
      ctx.fillStyle = p.color + (alpha * 0.25) + ')';
      ctx.fill();
    }

    // Draw molecular bonds
    for (var i = 0; i < particles.length; i++) {
      for (var j = i + 1; j < particles.length; j++) {
        var p1 = particles[i];
        var p2 = particles[j];
        var dx = p1.x - p2.x;
        var dy = p1.y - p2.y;
        var dist = Math.sqrt(dx * dx + dy * dy);

        var maxDist = state === 'orbiting' ? 90 : 80;

        if (dist < maxDist) {
          var opacity = (1 - dist / maxDist) * 0.22;
          if (state === 'syncing') {
            var elapsed = Math.min((time - syncStartTime) / 1800, 1);
            opacity += (1 - opacity) * Math.sin(elapsed * Math.PI) * 0.15;
          }
          ctx.beginPath();
          ctx.moveTo(p1.x, p1.y);
          ctx.lineTo(p2.x, p2.y);
          ctx.strokeStyle = 'rgba(230, 162, 164, ' + opacity + ')';
          ctx.lineWidth = 0.6;
          ctx.stroke();
        }
      }
    }

    requestAnimationFrame(draw);
  }
  draw();
})();
</script>
